public static async Task<IActionResult> Run(
    [HttpTrigger(AuthorizationLevel.Function, "get", "post")] HttpRequest req,
    ILogger log)

    string name = req.Query["name"];
    return new OkObjectResult($"Hello, {name}!");
}
